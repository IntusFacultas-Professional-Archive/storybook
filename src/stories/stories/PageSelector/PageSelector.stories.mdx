# Page Selector

```html
<template>
<IFCThemeProvider >
  <IFCPageSelector @change="page = $event" :value="page" :pages="pages" />
</IFCThemeProvider>
</template>

<script>
import { IFCThemeProvider, IFCPageSelector } from '@intusfacultas/component-library/build/index.esm';

export default {
  components: {
    IFCThemeProvider,
    IFCPageSelector,
  },
  data() {
    return {
      page: 1,
      pages: 20,
    };
  },
};
</script>

```
