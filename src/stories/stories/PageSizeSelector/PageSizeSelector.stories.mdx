# Page Selector

```html
<template>
<IFCThemeProvider >
  <IFCPageSizeSelector v-model="page" :pageSizes="pageSizes" />
</IFCThemeProvider>
</template>

<script>
import {
  IFCThemeProvider, IFCPageSizeSelector
} from '@nasic/component-library/build/index.esm';

export default {
  components: {
    IFCThemeProvider,
    IFCPageSizeSelector
  },
  data() {
    return {
      page: 1,
      pageSizes: [1,5,10,20]
    }
  }
}
</script>
```
